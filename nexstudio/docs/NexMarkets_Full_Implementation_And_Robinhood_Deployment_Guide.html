1. Purpose, boundaries and non-negotiables

This document is the implementation source of truth. It converts the approved NexMarkets product experience into deployable frontend, backend, worker and EVM architecture. docs/NexMarkets_NexCard_Exact_Export.html is a one-time visual-and-interaction reference only: copy its approved UI into route and component source files, then the application must work after that reference file is deleted. Never serve, import, iframe, inject or otherwise render the export at runtime. Every visible record and primary action must use authenticated, Prisma-backed API data and durable workflows; unavailable infrastructure must produce an explicit error, never fabricated, seeded or simulated account data. Prisma owns persistence, Hardhat owns the EVM toolchain, and HyperFrames renders through HeyGen.

What NexMarkets must do

  • Let guests explore Studio, Marketplace, public NexCards and public profiles before signing in.
  • Keep Creation and Reputation as two separate NexMind experiences.
  • Accept websites, pasted announcements, tweets, social copy, files, Project Sources and Marketplace work as creation sources.
  • Charge 5 USDC for video, or 4 USDC when the payer wallet holds at least 50,000 NEX at payment confirmation.
  • Charge 0.10 USDC for an infographic or still image.
  • Create a free base NexCard after X connection. Require at least 50,000 NEX for live NexMind reputation enhancement.
  • Let production continue after the user leaves. Deliver in-app and Telegram notifications.
  • Support a delegated live briefing worker who speaks for a founder with explicit permissions.
  • Keep every payment, approval, revision, version and permission auditable.
  • Deploy economic contracts on Robinhood Chain. Keep personal data, transcripts and media off-chain.

What must never be placed on-chain

Do not write X data, names, biographies, transcripts, prompts, private source files, Telegram identifiers, media URLs or raw project context to a public chain. Contracts receive opaque identifiers, content hashes, wallet addresses, amounts and lifecycle state only.

Reference-export and data integrity rule

Break the approved export into typed page and component files while preserving its approved markup, layout, styles and interactions. The exported HTML must not appear in a Next.js route, response body, iframe, script injection, static import or runtime filesystem read. Replace every simulation with an API-backed state, mutation, durable job or explicit empty/error state. Test the application after removing the reference export from a copy of the workspace.

Release safety

The Solidity in this guide is production-oriented starter code, not an audit. Run unit tests, invariant tests, fuzz tests, a professional smart-contract audit, a legal review and a staged testnet launch before Robinhood Chain mainnet. Never copy private keys into source code or CI logs.

export function assertProductionTransition(from: ProductionStatus, to: ProductionStatus) {
  if (!allowed[from].includes(to)) throw new Error(`Invalid production transition: ${from} -> ${to}`);
}

Idempotency rule

Every payment, render, Telegram delivery and external webhook uses a stable idempotency key. The API stores the key with the result. A repeated request returns the original result rather than executing again.

Server-side quote pseudocode

export async function createProductionQuote(input: {
  productionId: string;
  payer: Address;
}) {
  const production = await productionRepo.getForUpdate(input.productionId);
  assert(production.status === "DIRECTION_READY");

  const config = await pricingConfigRepo.current();
  const [usdcBalance, nexBalance] = await Promise.all([
    publicClient.readContract({ address: config.usdc, abi: erc20Abi, functionName: "balanceOf", args: [input.payer] }),
    publicClient.readContract({ address: config.nex, abi: erc20Abi, functionName: "balanceOf", args: [input.payer] })
  ]);

  const standard = production.kind === "VIDEO" ? config.videoPrice : config.infographicPrice;
  const eligible = production.kind === "VIDEO" && nexBalance >= config.nexThreshold;
  const finalPrice = eligible ? config.videoDiscountPrice : standard;

  return quoteRepo.create({
    productionId: production.id,
    payer: input.payer,
    standard,
    finalPrice,
    eligible,
    usdcBalance,
    expiresAt: addMinutes(new Date(), 10)
  });
}

export function createAudioFeatures(stream: MediaStream) {
  const context = new AudioContext();
  const source = context.createMediaStreamSource(stream);
  const analyser = context.createAnalyser();
  analyser.fftSize = 1024;
  analyser.smoothingTimeConstant = 0.78;
  source.connect(analyser);

  const bins = new Uint8Array(analyser.frequencyBinCount);
  const time = new Uint8Array(analyser.fftSize);

  function sample() {
    analyser.getByteFrequencyData(bins);
    analyser.getByteTimeDomainData(time);
    let rms = 0;
    for (const value of time) {
      const centred = (value - 128) / 128;
      rms += centred * centred;
    }
    rms = Math.sqrt(rms / time.length);
    const band = (from: number, to: number) => {
      let sum = 0;
      for (let i = from; i < to; i++) sum += bins[i];
      return sum / Math.max(1, to - from) / 255;
    };
    return { rms, low: band(1, 12), mid: band(12, 90), high: band(90, 260) };
  }
  return { context, analyser, sample, close: () => context.close() };
}

import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";

contract NexPricingRegistry is AccessControl {
    bytes32 public constant CONFIG_ROLE = keccak256("CONFIG_ROLE");

    error ZeroAddress();
    error InvalidPrice();

    IERC20 public usdc;
    IERC20 public nex;
    address public treasury;
    uint256 public videoPrice;
    uint256 public discountedVideoPrice;
    uint256 public infographicPrice;
    uint256 public nexThreshold;
    uint64 public configVersion;

    event ConfigurationUpdated(
        uint64 indexed version,
        address indexed usdc,
        address indexed nex,
        address treasury,
        uint256 videoPrice,
        uint256 discountedVideoPrice,
        uint256 infographicPrice,
        uint256 nexThreshold
    );

    constructor(address admin, address configManager) {
        if (admin == address(0) || configManager == address(0)) revert ZeroAddress();
        _grantRole(DEFAULT_ADMIN_ROLE, admin);
        _grantRole(CONFIG_ROLE, configManager);
    }

    function configure(
        IERC20 usdc_,
        IERC20 nex_,
        address treasury_,
        uint256 videoPrice_,
        uint256 discountedVideoPrice_,
        uint256 infographicPrice_,
        uint256 nexThreshold_
    ) external onlyRole(CONFIG_ROLE) {
        if (address(usdc_) == address(0) || address(nex_) == address(0) || treasury_ == address(0)) {
            revert ZeroAddress();
        }
        if (
            videoPrice_ == 0 || discountedVideoPrice_ == 0 || infographicPrice_ == 0 ||
            discountedVideoPrice_ > videoPrice_ || nexThreshold_ == 0
        ) revert InvalidPrice();

        usdc = usdc_;
        nex = nex_;
        treasury = treasury_;
        videoPrice = videoPrice_;
        discountedVideoPrice = discountedVideoPrice_;
        infographicPrice = infographicPrice_;
        nexThreshold = nexThreshold_;
        unchecked { configVersion += 1; }

        emit ConfigurationUpdated(
            configVersion, address(usdc_), address(nex_), treasury_, videoPrice_,
            discountedVideoPrice_, infographicPrice_, nexThreshold_
        );
    }

    function isNexEligible(address account) public view returns (bool) {
        return nex.balanceOf(account) >= nexThreshold;
    }

    function quoteVideo(address account) external view returns (uint256 amount, bool eligible, uint64 version) {
        eligible = isNexEligible(account);
        amount = eligible ? discountedVideoPrice : videoPrice;
        version = configVersion;
    }

    function quoteInfographic() external view returns (uint256 amount, uint64 version) {
        return (infographicPrice, configVersion);
    }
}

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.28;

import {AccessControl} from "@openzeppelin/contracts/access/AccessControl.sol";
import {Pausable} from "@openzeppelin/contracts/utils/Pausable.sol";
import {ReentrancyGuard} from "@openzeppelin/contracts/utils/ReentrancyGuard.sol";
import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import {SafeERC20} from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
import {NexPricingRegistry} from "./NexPricingRegistry.sol";

contract NexProductionPayments is AccessControl, Pausable, ReentrancyGuard {
    using SafeERC20 for IERC20;

    bytes32 public constant OPERATOR_ROLE = keccak256("OPERATOR_ROLE");
    bytes32 public constant PAUSER_ROLE = keccak256("PAUSER_ROLE");

    enum Kind { Video, Infographic }
    enum Status { None, Paid, Settled, Refunded }

    struct Payment {
        address payer;
        uint128 amount;
        uint64 paidAt;
        uint64 configVersion;
        Kind kind;
        Status status;
    }

    NexPricingRegistry public immutable registry;
    mapping(bytes32 => Payment) public payments;

    error ZeroAddress();
    error AlreadyExists();
    error UnknownPayment();
    error InvalidStatus();
    error StaleQuote();
    error PriceChanged(uint256 expected, uint256 actual);

    event ProductionPaid(
        bytes32 indexed productionId,
        address indexed payer,
        Kind kind,
        uint256 amount,
        bool nexEligible,
        uint64 configVersion
    );
    event ProductionSettled(bytes32 indexed productionId, uint256 amount, address indexed treasury);
    event ProductionRefunded(bytes32 indexed productionId, uint256 amount, address indexed payer, bytes32 reasonHash);

    constructor(address admin, address operator, NexPricingRegistry registry_) {
        if (admin == address(0) || operator == address(0) || address(registry_) == address(0)) revert ZeroAddress();
        registry = registry_;
        _grantRole(DEFAULT_ADMIN_ROLE, admin);
        _grantRole(OPERATOR_ROLE, operator);
        _grantRole(PAUSER_ROLE, admin);
    }

    function pay(
        bytes32 productionId,
        Kind kind,
        uint256 expectedAmount,
        uint64 expectedConfigVersion
    ) external nonReentrant whenNotPaused {
        if (payments[productionId].status != Status.None) revert AlreadyExists();

        uint256 amount;
        bool eligible;
        uint64 version;
        if (kind == Kind.Video) {
            (amount, eligible, version) = registry.quoteVideo(msg.sender);
        } else {
            (amount, version) = registry.quoteInfographic();
        }
        if (version != expectedConfigVersion) revert StaleQuote();
        if (amount != expectedAmount) revert PriceChanged(expectedAmount, amount);

        IERC20 token = registry.usdc();
        token.safeTransferFrom(msg.sender, address(this), amount);

        payments[productionId] = Payment({
            payer: msg.sender,
            amount: uint128(amount),
            paidAt: uint64(block.timestamp),
            configVersion: version,
            kind: kind,
            status: Status.Paid
        });
        emit ProductionPaid(productionId, msg.sender, kind, amount, eligible, version);
    }

    function settle(bytes32 productionId) external onlyRole(OPERATOR_ROLE) nonReentrant {
        Payment storage payment = payments[productionId];
        if (payment.status == Status.None) revert UnknownPayment();
        if (payment.status != Status.Paid) revert InvalidStatus();
        payment.status = Status.Settled;
        address treasury = registry.treasury();
        registry.usdc().safeTransfer(treasury, payment.amount);
        emit ProductionSettled(productionId, payment.amount, treasury);
    }

    function refund(bytes32 productionId, bytes32 reasonHash) external onlyRole(OPERATOR_ROLE) nonReentrant {
        Payment storage payment = payments[productionId];
        if (payment.status == Status.None) revert UnknownPayment();
        if (payment.status != Status.Paid) revert InvalidStatus();
        payment.status = Status.Refunded;
        registry.usdc().safeTransfer(payment.payer, payment.amount);
        emit ProductionRefunded(productionId, payment.amount, payment.payer, reasonHash);
    }

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.28;

import {AccessControl} from "@openzeppelin/contracts/access/AccessControl.sol";
import {Pausable} from "@openzeppelin/contracts/utils/Pausable.sol";
import {ReentrancyGuard} from "@openzeppelin/contracts/utils/ReentrancyGuard.sol";
import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import {SafeERC20} from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";

contract NexWorkEscrow is AccessControl, Pausable, ReentrancyGuard {
    using SafeERC20 for IERC20;

    bytes32 public constant OPERATOR_ROLE = keccak256("OPERATOR_ROLE");
    bytes32 public constant RESOLVER_ROLE = keccak256("RESOLVER_ROLE");
    bytes32 public constant PAUSER_ROLE = keccak256("PAUSER_ROLE");

    enum Status { None, Funded, InProgress, Delivered, Revision, Approved, Released, Disputed, Refunded, Cancelled }

    struct Escrow {
        address founder;
        address worker;
        uint128 amount;
        uint64 fundedAt;
        uint64 deliveredAt;
        uint64 reviewDeadline;
        uint32 revisionCount;
        bool autoRelease;
        Status status;
    }

    IERC20 public immutable usdc;
    address public feeTreasury;
    uint16 public platformFeeBps;
    mapping(bytes32 => Escrow) public escrows;

    error ZeroAddress();
    error InvalidAmount();
    error AlreadyExists();
    error UnknownEscrow();
    error InvalidStatus();
    error NotFounder();
    error NotWorker();
    error ReviewWindowOpen();
    error FeeTooHigh();

    event EscrowFunded(bytes32 indexed workroomId, address indexed founder, address indexed worker, uint256 amount, bool autoRelease);
    event WorkStarted(bytes32 indexed workroomId);
    event DeliverySubmitted(bytes32 indexed workroomId, bytes32 indexed deliveryHash, uint64 reviewDeadline);
    event RevisionRequested(bytes32 indexed workroomId, bytes32 indexed requestHash, uint32 revisionCount);
    event DeliveryApproved(bytes32 indexed workroomId);
    event PaymentReleased(bytes32 indexed workroomId, uint256 workerAmount, uint256 feeAmount);
    event DisputeOpened(bytes32 indexed workroomId, bytes32 indexed reasonHash);
    event DisputeResolved(bytes32 indexed workroomId, uint256 founderAmount, uint256 workerAmount, uint256 feeAmount);
    event EscrowRefunded(bytes32 indexed workroomId, uint256 amount);

    constructor(address admin, address operator, address resolver, IERC20 usdc_, address feeTreasury_, uint16 platformFeeBps_) {
        if (admin == address(0) || operator == address(0) || resolver == address(0) || address(usdc_) == address(0) || feeTreasury_ == address(0)) revert ZeroAddress();
        if (platformFeeBps_ > 1000) revert FeeTooHigh();
        usdc = usdc_;
        feeTreasury = feeTreasury_;
        platformFeeBps = platformFeeBps_;
        _grantRole(DEFAULT_ADMIN_ROLE, admin);
        _grantRole(OPERATOR_ROLE, operator);
        _grantRole(RESOLVER_ROLE, resolver);
        _grantRole(PAUSER_ROLE, admin);
    }

    function fund(bytes32 workroomId, address worker, uint256 amount, bool autoRelease) external nonReentrant whenNotPaused {
        if (worker == address(0)) revert ZeroAddress();
        if (amount == 0 || amount > type(uint128).max) revert InvalidAmount();
        if (escrows[workroomId].status != Status.None) revert AlreadyExists();
        usdc.safeTransferFrom(msg.sender, address(this), amount);
        escrows[workroomId] = Escrow({
            founder: msg.sender,
            worker: worker,
            amount: uint128(amount),
            fundedAt: uint64(block.timestamp),
            deliveredAt: 0,
            reviewDeadline: 0,
            revisionCount: 0,
            autoRelease: autoRelease,
            status: Status.Funded
        });
        emit EscrowFunded(workroomId, msg.sender, worker, amount, autoRelease);
    }

    function start(bytes32 workroomId) external onlyRole(OPERATOR_ROLE) {
        Escrow storage e = _escrow(workroomId);
        if (e.status != Status.Funded) revert InvalidStatus();
        e.status = Status.InProgress;
        emit WorkStarted(workroomId);
    }

    function submitDelivery(bytes32 workroomId, bytes32 deliveryHash, uint64 reviewSeconds) external whenNotPaused {
        Escrow storage e = _escrow(workroomId);
        if (msg.sender != e.worker && !hasRole(OPERATOR_ROLE, msg.sender)) revert NotWorker();
        if (e.status != Status.InProgress && e.status != Status.Revision) revert InvalidStatus();
        e.status = Status.Delivered;
        e.deliveredAt = uint64(block.timestamp);
        e.reviewDeadline = uint64(block.timestamp) + reviewSeconds;
        emit DeliverySubmitted(workroomId, deliveryHash, e.reviewDeadline);
    }

    function requestRevision(bytes32 workroomId, bytes32 requestHash) external {
        Escrow storage e = _escrow(workroomId);
        if (msg.sender != e.founder) revert NotFounder();
        if (e.status != Status.Delivered) revert InvalidStatus();
        e.status = Status.Revision;
        unchecked { e.revisionCount += 1; }
        emit RevisionRequested(workroomId, requestHash, e.revisionCount);
    }

    function approve(bytes32 workroomId) external {
        Escrow storage e = _escrow(workroomId);
        if (msg.sender != e.founder) revert NotFounder();
        if (e.status != Status.Delivered) revert InvalidStatus();
        e.status = Status.Approved;
        emit DeliveryApproved(workroomId);
    }

    function release(bytes32 workroomId) external nonReentrant {
        Escrow storage e = _escrow(workroomId);
        bool operator = hasRole(OPERATOR_ROLE, msg.sender);
        bool founderApproved = e.status == Status.Approved && (msg.sender == e.founder || operator);
        bool autoReleaseReady = e.autoRelease && e.status == Status.Delivered && block.timestamp >= e.reviewDeadline;
        if (!founderApproved && !autoReleaseReady) {
            if (e.status == Status.Delivered && block.timestamp < e.reviewDeadline) revert ReviewWindowOpen();
            revert InvalidStatus();
        }
        e.status = Status.Released;
        (uint256 workerAmount, uint256 feeAmount) = _split(e.amount);
        usdc.safeTransfer(e.worker, workerAmount);
        if (feeAmount > 0) usdc.safeTransfer(feeTreasury, feeAmount);
        emit PaymentReleased(workroomId, workerAmount, feeAmount);
    }

    function openDispute(bytes32 workroomId, bytes32 reasonHash) external {
        Escrow storage e = _escrow(workroomId);
        if (msg.sender != e.founder && msg.sender != e.worker) revert InvalidStatus();
        if (e.status != Status.Delivered && e.status != Status.Revision && e.status != Status.InProgress) revert InvalidStatus();
        e.status = Status.Disputed;
        emit DisputeOpened(workroomId, reasonHash);
    }

    function resolveDispute(bytes32 workroomId, uint256 founderAmount, uint256 workerGrossAmount) external onlyRole(RESOLVER_ROLE) nonReentrant {
        Escrow storage e = _escrow(workroomId);
        if (e.status != Status.Disputed) revert InvalidStatus();
        if (founderAmount + workerGrossAmount != e.amount) revert InvalidAmount();
        e.status = Status.Released;
        uint256 feeAmount = (workerGrossAmount * platformFeeBps) / 10_000;
        uint256 workerAmount = workerGrossAmount - feeAmount;
        if (founderAmount > 0) usdc.safeTransfer(e.founder, founderAmount);
        if (workerAmount > 0) usdc.safeTransfer(e.worker, workerAmount);
        if (feeAmount > 0) usdc.safeTransfer(feeTreasury, feeAmount);
        emit DisputeResolved(workroomId, founderAmount, workerAmount, feeAmount);
    }

    function refundBeforeStart(bytes32 workroomId) external nonReentrant {
        Escrow storage e = _escrow(workroomId);
        if (msg.sender != e.founder && !hasRole(OPERATOR_ROLE, msg.sender)) revert NotFounder();
        if (e.status != Status.Funded) revert InvalidStatus();
        e.status = Status.Refunded;
        usdc.safeTransfer(e.founder, e.amount);
        emit EscrowRefunded(workroomId, e.amount);
    }

    function setFee(address treasury, uint16 feeBps) external onlyRole(DEFAULT_ADMIN_ROLE) {
        if (treasury == address(0)) revert ZeroAddress();
        if (feeBps > 1000) revert FeeTooHigh();
        feeTreasury = treasury;
        platformFeeBps = feeBps;
    }

    function pause() external onlyRole(PAUSER_ROLE) { _pause(); }
    function unpause() external onlyRole(PAUSER_ROLE) { _unpause(); }

    function _escrow(bytes32 id) internal view returns (Escrow storage e) {
        e = escrows[id];
        if (e.status == Status.None) revert UnknownEscrow();
    }

    function _split(uint256 gross) internal view returns (uint256 workerAmount, uint256 feeAmount) {
        feeAmount = (gross * platformFeeBps) / 10_000;
        workerAmount = gross - feeAmount;
    }
}

Why opaque IDs are bytes32

Generate keccak256(abi.encodePacked("NEX:PRODUCTION:", productionUuid)) or NEX:WORKROOM on the server. Store the mapping in the database. This prevents raw internal IDs and content from being exposed while preserving deterministic reconciliation.

Application security
